for i in l:
    q1=df[i].quantile(0.25)
    q2=df[i].quantile(0.5)
    q3=df[i].quantile(0.75)
    iqr = q3 - q1
    upper_lm =  q3 + ( 1.5 * iqr )
    lower_lm = q1 - ( 1.5 * iqr )
    df[i]=np.where(df[i]>upper_lm,upper_lm,np.where(df[i]<lower_lm,lower_lm,df[i]))
